Quick Start
import requests # Full agent analysis res = requests.post( "http://localhost:8000/analyze", json={"text": "Click here to win!"} ) report = res.json() print(report["detected_intent"]) print(report["risk_assessment"]) print(report["graph_execution"])curl -X POST http://localhost:8000/analyze \ -H "Content-Type: application/json" \ -d '{"text": "Meeting at 3pm tomorrow"}' # Quick classify endpoint curl -X POST http://localhost:8000/classify \ -H "Content-Type: application/json" \ -d '{"text": "You won a free car!"}'# Stream node-by-node execution import requests, json res = requests.post( "http://localhost:8000/analyze/stream", json={"text": "Urgent offer!"}, stream=True ) for line in res.iter_lines(): node = json.loads(line) print(f"✓ {node['node']}")
